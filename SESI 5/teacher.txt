public class Teacher extends Person {
    private int numCourses = 0;
    private String[] courses = new String[10]; // Asumsi maksimal 10 mata kuliah

    // Constructor
    public Teacher(String name, String address) {
        super(name, address);
    }

    @Override
    public String toString() {
        return "Teacher: " + super.toString();
    }

    // Menambahkan mata kuliah baru
    public boolean addCourse(String course) {
        // Cek apakah mata kuliah sudah ada
        for (int i = 0; i < numCourses; i++) {
            if (courses[i].equalsIgnoreCase(course)) {
                return false; // Gagal, course sudah ada
            }
        }
        // Tambahkan jika belum ada dan slot masih tersedia
        if (numCourses < 10) {
            courses[numCourses] = course;
            numCourses++;
            return true;
        }
        return false;
    }

    // Menghapus mata kuliah
    public boolean removeCourse(String course) {
        boolean found = false;
        int courseIndex = -1;
        
        // Cari index mata kuliah
        for (int i = 0; i < numCourses; i++) {
            if (courses[i].equalsIgnoreCase(course)) {
                found = true;
                courseIndex = i;
                break;
            }
        }

        // Jika ketemu, geser elemen array untuk menutupi yang dihapus
        if (found) {
            for (int i = courseIndex; i < numCourses - 1; i++) {
                courses[i] = courses[i + 1];
            }
            courses[numCourses - 1] = null;
            numCourses--;
            return true;
        }
        
        return false; // Gagal, course tidak ditemukan
    }
}
